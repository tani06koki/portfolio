install_if_missing <- function(package) {
  if (!require(package, character.only = TRUE)) {
    install.packages(package, dependencies = TRUE)
    library(package, character.only = TRUE)
  }
}

1.2 Data Preparation

Show the code 
# Data Preparation
minDate <- min(data$Date)

Join <- left_join(data, sector, by="Symbol", relationship = "many-to-many")

Data <- Join %>%
  select(c(Date, Symbol, Name, Sector, Close)) %>%
  mutate(Return = (Close - lag(Close)) / lag(Close)) %>%
  filter(Date != minDate)

1.3 Calculation of ratios

1.3.1 Sharpe Ratio

Show the code 
Sharpe <- Data %>%
  group_by(Symbol) %>%
  mutate(Sector = Sector) %>%
  summarise(
    sharpeRatio = mean(Return) / sd(Return),
    Sector = min(Sector)
            )

Sharpe_sector <- Sharpe %>%
  group_by(Sector) %>%
  arrange(desc(sharpeRatio)) %>%
  slice_head(n = 6) %>%
  filter(sharpeRatio > 0) %>%
  arrange(desc(sharpeRatio))%>%
  ungroup()

1.3.2 RSI

Show the code 
# RSI - 14 days
# Remove currently 
RSI_70_below <- Data %>%
  filter(Symbol %in% Sharpe_sector$Symbol) %>%
  group_by(Symbol)%>%
  mutate(RSI = RSI(Close, n = 14)) %>%
  filter(Date == max(Date)) %>%
  filter(RSI < 70)

1.3.3 EMA

Show the code 
# EMA
EMA <- Data %>%
  group_by(Symbol) %>%
  mutate(
    ema_fast = EMA(Close, n = 12),
    ema_slow = EMA(Close, n = 26),
    macd_line = ema_fast - ema_slow,
    signal_line = EMA(macd_line, n = 9),
    histogram = macd_line - signal_line,
    RSI = RSI(Close, n = 14)
  )

# change the data type as Date
EMA$Date <- as.Date(EMA$Date)

EMA_RSI <- EMA %>%
filter(Symbol %in% RSI_70_below$Symbol)
